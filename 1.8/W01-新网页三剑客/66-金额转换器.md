<!DOCTYPE html>
<html lang="en">

	<head>
		<meta charset="UTF-8">
		<meta name="viewport" content="width=device-width, initial-scale=1.0">
		<title>金额转换器</title>
		<style>
			.box {
				width: 250px;
				margin: 100px auto;
				position: relative;
			}



			.lastBox {
				margin-top: 15px;
			}
			
			#result{
				color: red;
			}
		</style>
	</head>

	<body>
		<div class="box">
			请输入整数数字： 
			<input type="text" id="numberInput">


			<div class="lastBox">
				转换结果：<p id="result"></p>
			</div>


		</div>
		<script>
			const numberInput = document.getElementById('numberInput');
			const resultElement = document.getElementById('result');

			numberInput.addEventListener('keyup', function() {
				const input = numberInput.value.trim();
				// 验证输入是否为有效数字
				if (/^-?\d+$/.test(input)) {
					// 转换数字为大写人民币
					const result = digitToChinese(input);
					// 显示结果
					resultElement.innerText = `人民币${result}元整`;
				} else {
					// 输入无效时清除结果
					resultElement.innerText = '输入数字无效';
				}
			});
			const map = {
				零: '零',
				一: '壹',
				二: '贰',
				三: '叁',
				四: '肆',
				五: '伍',
				六: '陆',
				七: '柒',
				八: '捌',
				九: '玖',
				十: '拾',
				百: '佰',
				千: '仟',
				万: '萬',
				亿: '亿',
			}
			const bigUnits = ["", "万", "亿"];
			const chars = ["零", "一", "二", "三", "四", "五", "六", "七", "八", "九"];
			const units = ["", "十", "百", "千"];
			// 数字转大写人民币的函数
			function digitToChinese(num) {

				// 四位进行分割
				const numStr = num.toString().replace(/(?=(\d{4})+$)/g, ',')
					.split(',')
					.filter(Boolean)

				let result = ''
				for (let i = 0; i < numStr.length; i++) {
					const part = numStr[i];
					const c = _transform(part)
					let u = bigUnits[numStr.length - i - 1];
					if (c === chars[0]) {
						u = ''
					}
					result += c + u
				}
				result = _handleZero(result)
				// 处理一十情况
				if (result === '一十') {
					result = '十'
				}

				return result.split('').map(s => map[s]).join('')
			}

			function _handleZero(str) {
				return str.replace(/零{2,}/g, '零').replace(/零+$/, '')
			}

			function _transform(n) {
				if (n === '0000') {
					return chars[0]
				}

				let result = ''
				for (let i = 0; i < n.length; i++) {
					// 转换汉字
					const c = chars[+n[i]];
					// 加单位 得到单位
					let u = units[n.length - 1 - i]
					if (c === chars[0]) {
						u = ''
					}
					result += c + u
				}
			 return _handleZero(result)
			}
		</script>
	</body>

</html>
