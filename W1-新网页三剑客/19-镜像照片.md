```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <style>
      img {
        width: 100px;
        height: 150px;
        border: 1px solid #fff;
        border-radius: 20px;
        transition: transform 1s ease-in-out; /*ease-in-out 慢速开始和结束*/
      }
      /*找到 img 的第二个兄弟元素*/
      img:nth-child(2) {
        transform: scaleY(-1); /*图片沿 Y 轴向下翻转*/
      }
      /*找到 img 的第三个兄弟元素*/
      img:nth-child(3) {
        transform: scaleX(-1); /*图片沿 X 轴向左翻转*/
      }
      /*找到 img 的第四个兄弟元素*/
      img:nth-child(4) {
        transform: scale(-1); /*图片沿 Y 轴向下翻转，又沿 X 轴向左翻转*/
      }
      /*鼠标放置图片上翻转图片会变成原来的方向*/
      img:hover {
        transform: scale(1);
      }
    </style>
  </head>
  <body>
    <!--正方向-->
    <img src="aneta.jpg" />
    <!--沿 Y 轴向下翻-->
    <img src="aneta.jpg" />
    <!--沿 X 轴向左翻-->
    <img src="aneta.jpg" />
    <!--沿 Y 轴向下翻并且沿 X 轴向左翻-->
    <img src="aneta.jpg" />
  </body>
</html>
```
