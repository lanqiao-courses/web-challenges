```js
function useCategoryEffect() {
  var type = "商品种类";

  var add = function () {
    console.log("我在添加:" + type);
  };

  var update = function () {
    console.log("我在修改:" + type);
  };

  return { add, update };
}
```

```js
function useShoppingCartEffect() {
  var type = "商品";

  var add = function () {
    console.log("我在往购物车中添加:" + type);
  };

  var update = function () {
    console.log("我在修改购物车中的:" + type);
  };

  return { add, update };
}
```
