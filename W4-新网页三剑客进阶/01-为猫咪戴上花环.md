```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>

  <body>
    <canvas
      id="myCanvas"
      width="200"
      height="200"
      style="border:1px solid #000;"
    >
      您的浏览器不支持 HTML5 canvas 标签。
    </canvas>
    <input type="button" id="btnChange" value="美颜" />
  </body>
  <script>
    //1. 得到Canvas对象
    var mycanvas = document.getElementById("myCanvas");
    //2. 得到一个CanvasRenderingContext2D对象
    var ctx = mycanvas.getContext("2d");
    //3. 创建图片对象
    var img = new Image();
    img.src = "https://labfile.oss.aliyuncs.com/courses/3944/cat.jpg";
    img.onload = function () {
      //4. 绘制图像
      ctx.drawImage(img, 0, 0, 150, 150);
    };
    //5. 单击“美颜”按钮
    document.getElementById("btnChange").onclick = function () {
      //6. 创建花环图片对象
      var img = new Image();
      img.src = "https://labfile.oss.aliyuncs.com/courses/3944/border.png";
      //7. 绘制图像
      img.onload = function () {
        ctx.drawImage(img, 0, 0, 150, 150);
      };
    };
  </script>
</html>
```
